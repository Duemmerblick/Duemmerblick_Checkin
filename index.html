function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_BUCHUNGEN); var data = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < data.length; i++) { if(!data[i][0]) continue; result.push({ id: data[i][0], wohnung: data[i][1], name: data[i][2], anreise: Utilities.formatDate(new Date(data[i][3]), "GMT+1", "yyyy-MM-dd"), abreise: Utilities.formatDate(new Date(data[i][4]), "GMT+1", "yyyy-MM-dd"), anschrift: data[i][5], telefon: data[i][6], email: data[i][7], ausweis: data[i][8], signature: data[i][9] }); } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var params = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // UNTERSCHEIDUNG: Notiz oder Buchung? if (params.type === "note") { var nSheet = ss.getSheetByName(SHEET_NOTIZEN) || ss.insertSheet(SHEET_NOTIZEN); if (nSheet.getLastRow() === 0) { nSheet.appendRow(["Zeitstempel", "Wohnung", "Notiz"]); } nSheet.appendRow([new Date(), params.wohnung, params.text]); return ContentService.createTextOutput("Notiz gespeichert"); } // Normaler Buchungs-Update-Teil var sheet = ss.getSheetByName(SHEET_BUCHUNGEN); var data = sheet.getDataRange().getValues(); var id = params.id; var rowIndex = -1; for (var i = 1; i < data.length; i++) { if (data[i][0].toString() == id.toString()) { rowIndex = i + 1; break; } } if (params.delete === true && rowIndex > -1) { sheet.deleteRow(rowIndex); return ContentService.createTextOutput("Gelöscht"); } var rowData = [id, params.wohnung, params.name, params.anreise, params.abreise, params.anschrift || "", params.telefon || "", params.email || "", params.ausweis || "", params.signature || ""]; if (rowIndex > -1) { sheet.getRange(rowIndex, 1, 1, 10).setValues([rowData]); } else { sheet.appendRow(rowData); } return ContentService.createTextOutput("Erfolg"); } function importMoreBookings() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_BUCHUNGEN); var data = sheet.getDataRange().getValues(); var existingIds = data.map(function(row) { return row[0].toString(); }); processIcal(URL_WOHNUNG_1, "Dümmerblick 1", sheet, existingIds); processIcal(URL_WOHNUNG_2, "Dümmerblick 2", sheet, existingIds); } function processIcal(url, wohnungsName, sheet, existingIds) { var response = UrlFetchApp.fetch(url); var icalContent = response.getContentText(); var events = icalContent.split("BEGIN:VEVENT"); for (var i = 1; i < events.length; i++) { var ev = events[i]; var uid = extractIcalField(ev, "UID"); if (uid && existingIds.indexOf(uid) === -1) { var summary = extractIcalField(ev, "SUMMARY"); var description = extractIcalField(ev, "DESCRIPTION"); var startStr = extractIcalField(ev, "DTSTART"); var endStr = extractIcalField(ev, "DTEND"); let gastName = ""; let nameMatch = description.match(/Name:\s*([^\n\r]+)/i) || description.match(/Guest:\s*([^\n\r]+)/i); if (nameMatch && nameMatch[1] && !nameMatch[1].toLowerCase().includes("not available")) { gastName = nameMatch[1].trim(); } else { let cleanSummary = summary.replace("Buchung More-Bookings-Object", "").replace("Booking:", "").trim(); if (cleanSummary !== "" && !cleanSummary.toLowerCase().includes("not available")) { gastName = cleanSummary; } else { gastName = "Gast #" + uid.substring(uid.length - 5); } } var anreise = startStr.substring(0,4) + "-" + startStr.substring(4,6) + "-" + startStr.substring(6,8); var abreise = endStr.substring(0,4) + "-" + endStr.substring(4,6) + "-" + endStr.substring(6,8); sheet.appendRow([uid, wohnungsName, gastName, anreise, abreise, "", "", "", "", ""]); existingIds.push(uid); } } } function extractIcalField(content, field) { var regex = new RegExp(field + "(?:;[^:]*)?:(.*(?:\\r?\\n [^\\r\\n]*)*)", "i"); var match = regex.exec(content); if (match && match[1]) { return match[1].replace(/\r?\\n/g, " ").replace(/\r?\n /g, "").trim(); } return ""; }
