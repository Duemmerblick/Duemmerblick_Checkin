// ========================================== // DÜMMERBLICK MANAGER - BACKEND v9.1 (Wohnungs-Fix) // ========================================== var SHEET_NAME = "Buchungen"; var ICAL_URL = "https://api.more-bookings.com/api/ical_augmented_calendar?&object_id=ag1mZXdveWllbGRtZ210ch4LEhFjdXN0b21lcl9wcm9wZXJ0eRiAgMDVyrLECgw&file=calendar.ics"; function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); var data = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < data.length; i++) { if(!data[i][0]) continue; result.push({ id: data[i][0], wohnung: data[i][1], name: data[i][2], anreise: Utilities.formatDate(new Date(data[i][3]), "GMT+1", "yyyy-MM-dd"), abreise: Utilities.formatDate(new Date(data[i][4]), "GMT+1", "yyyy-MM-dd"), anschrift: data[i][5], telefon: data[i][6], email: data[i][7], ausweis: data[i][8], signature: data[i][9] }); } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var params = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); var data = sheet.getDataRange().getValues(); var id = params.id; var rowIndex = -1; for (var i = 1; i < data.length; i++) { if (data[i][0].toString() == id.toString()) { rowIndex = i + 1; break; } } if (params.delete === true) { if (rowIndex > -1) { sheet.deleteRow(rowIndex); return ContentService.createTextOutput("Gelöscht"); } return ContentService.createTextOutput("Nicht gefunden"); } var rowData = [id, params.wohnung, params.name, params.anreise, params.abreise, params.anschrift || "", params.telefon || "", params.email || "", params.ausweis || "", params.signature || ""]; if (rowIndex > -1) { sheet.getRange(rowIndex, 1, 1, 10).setValues([rowData]); } else { sheet.appendRow(rowData); } return ContentService.createTextOutput("Erfolg"); } function importMoreBookings() { var response = UrlFetchApp.fetch(ICAL_URL); var icalContent = response.getContentText(); var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); var data = sheet.getDataRange().getValues(); var existingIds = data.map(function(row) { return row[0].toString(); }); var events = icalContent.split("BEGIN:VEVENT"); for (var i = 1; i < events.length; i++) { var ev = events[i]; var uid = extractIcalField(ev, "UID"); if (!uid) continue; var summary = extractIcalField(ev, "SUMMARY"); var description = extractIcalField(ev, "DESCRIPTION"); var startStr = extractIcalField(ev, "DTSTART"); var endStr = extractIcalField(ev, "DTEND"); // VERSTÄRKTE WOHNUNGSERKENNUNG var fullText = (summary + " " + description).toUpperCase(); var wohnung = "Dümmerblick 1"; // Wir suchen nach " II" (mit Leerzeichen davor) oder " BLICK II" if (fullText.includes(" II") || fullText.includes("BLICK II") || fullText.includes("WOHNUNG II") || fullText.includes("UNIT II")) { wohnung = "Dümmerblick 2"; } // Namens-Extraktion var gastName = summary.replace(/Dümmerblick\s+[iI]+/gi, "").replace("Belegung:", "").replace("Booking:", "").trim(); if (gastName.toLowerCase().includes("booking.com") || /^\d+$/.test(gastName)) { var nameMatch = description.match(/Name:\s*([^\n\r]+)/i) || description.match(/Guest:\s*([^\n\r]+)/i); if (nameMatch && nameMatch[1]) { gastName = nameMatch[1].trim(); } } if (gastName === "" || /^\d+$/.test(gastName)) gastName = "Booking Gast"; var anreise = startStr.substring(0,4) + "-" + startStr.substring(4,6) + "-" + startStr.substring(6,8); var abreise = endStr.substring(0,4) + "-" + endStr.substring(4,6) + "-" + endStr.substring(6,8); var idx = existingIds.indexOf(uid); if (idx === -1) { sheet.appendRow([uid, wohnung, gastName, anreise, abreise, "", "", "", "", ""]); existingIds.push(uid); } else { var currentRow = idx + 1; // Falls die Wohnung in der Tabelle falsch ist (Dümmerblick 1 statt 2), korrigieren wir das hier mit! sheet.getRange(currentRow, 2).setValue(wohnung); var currentName = data[idx][2].toString(); if (currentName.toLowerCase().includes("booking") && !gastName.toLowerCase().includes("booking")) { sheet.getRange(currentRow, 3).setValue(gastName); } } } } function extractIcalField(content, field) { var regex = new RegExp(field + "(?:;[^:]*)?:(.*Internal(?:\\r?\\n [^\\r\\n]*)*)", "i"); // Kleine Korrektur am Regex für robustere Felder var match = content.match(new RegExp(field + ".*:(.*)")); return match ? match[1].trim() : ""; }
