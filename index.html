// ========================================== // DÜMMERBLICK MANAGER - BACKEND v12.1 // ========================================== var SHEET_BUCHUNGEN = "Buchungen"; var SHEET_NOTIZEN = "Notizen"; function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. BUCHUNGEN LADEN var bSheet = ss.getSheetByName(SHEET_BUCHUNGEN); var bData = bSheet.getDataRange().getValues(); var buchungen = []; for (var i = 1; i < bData.length; i++) { if(!bData[i][0]) continue; buchungen.push({ id: bData[i][0], wohnung: bData[i][1], name: bData[i][2], anreise: Utilities.formatDate(new Date(bData[i][3]), "GMT+1", "yyyy-MM-dd"), abreise: Utilities.formatDate(new Date(bData[i][4]), "GMT+1", "yyyy-MM-dd"), signature: bData[i][9] || "" }); } // 2. NOTIZEN LADEN (Nur offene) var notizen = []; var nSheet = ss.getSheetByName(SHEET_NOTIZEN); if (nSheet) { var nData = nSheet.getDataRange().getValues(); for (var j = nData.length - 1; j >= 1; j--) { // Notiz nur laden, wenn Spalte D NICHT "erledigt" ist if (nData[j][2] && nData[j][3] !== "erledigt") { var d = nData[j][0]; var zeitStr = (d instanceof Date) ? Utilities.formatDate(d, "GMT+1", "dd.MM. HH:mm") : d.toString(); notizen.push({ row: j + 1, zeit: zeitStr, wohnung: nData[j][1] || "Allgemein", text: nData[j][2].toString() }); } } } return ContentService.createTextOutput(JSON.stringify({ buchungen: buchungen, notizen: notizen })).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var params = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var nSheet = ss.getSheetByName(SHEET_NOTIZEN) || ss.insertSheet(SHEET_NOTIZEN); if (params.type === "note") { nSheet.appendRow([new Date(), params.wohnung, params.text, "offen"]); return ContentService.createTextOutput("OK"); } if (params.type === "done") { nSheet.getRange(params.row, 4).setValue("erledigt"); return ContentService.createTextOutput("OK"); } var bSheet = ss.getSheetByName(SHEET_BUCHUNGEN); var data = bSheet.getDataRange().getValues(); var rowIndex = -1; for (var i = 1; i < data.length; i++) { if (data[i][0].toString() == params.id.toString()) { rowIndex = i + 1; break; } } var rowData = [params.id, params.wohnung, params.name, params.anreise, params.abreise, "", "", "", "", params.signature || ""]; if (rowIndex > -1) { bSheet.getRange(rowIndex, 1, 1, 10).setValues([rowData]); } else { bSheet.appendRow(rowData); } return ContentService.createTextOutput("OK"); } function importMoreBookings() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_BUCHUNGEN); var data = sheet.getDataRange().getValues(); var tableIds = data.slice(1).map(function(row) { return row[0].toString(); }); var idsInIcal = []; var u1 = "https://api.more-bookings.com/api/ical_augmented_calendar?&object_id=ag1mZXdveWllbGRtZ210ch4LEhFjdXN0b21lcl9wcm9wZXJ0eRiAgMDVyrLECgw&file=calendar.ics"; var u2 = "https://api.more-bookings.com/api/ical_augmented_calendar?&object_id=ag1mZXdveWllbGRtZ210ch4LEhFjdXN0b21lcl9wcm9wZXJ0eRiAgIjzz8SlCQw&file=calendar.ics"; processEvents(UrlFetchApp.fetch(u1).getContentText().split("BEGIN:VEVENT"), "Dümmerblick 1", sheet, tableIds, idsInIcal); processEvents(UrlFetchApp.fetch(u2).getContentText().split("BEGIN:VEVENT"), "Dümmerblick 2", sheet, tableIds, idsInIcal); for (var j = tableIds.length - 1; j >= 0; j--) { if (idsInIcal.indexOf(tableIds[j]) === -1 && tableIds[j] !== "") { sheet.deleteRow(j + 2); } } } function processEvents(events, wohnung, sheet, tableIds, idsInIcal) { for (var i = 1; i < events.length; i++) { var ev = events[i]; var uid = extractIcalField(ev, "UID"); if (!uid) continue; idsInIcal.push(uid); if (tableIds.indexOf(uid) === -1) { var summary = extractIcalField(ev, "SUMMARY"); var desc = extractIcalField(ev, "DESCRIPTION"); var start = extractIcalField(ev, "DTSTART"); var end = extractIcalField(ev, "DTEND"); let gastName = ""; let nameMatch = desc.match(/Name:\s*([^\n\r]+)/i) || desc.match(/Guest:\s*([^\n\r]+)/i); if (nameMatch && nameMatch[1]) { gastName = nameMatch[1].trim(); } else { gastName = summary.replace("Buchung More-Bookings-Object", "").replace("Booking:", "").trim() || "Gast #" + uid.substring(uid.length - 5); } var d1 = start.substring(0,4) + "-" + start.substring(4,6) + "-" + start.substring(6,8); var d2 = end.substring(0,4) + "-" + end.substring(4,6) + "-" + end.substring(6,8); sheet.appendRow([uid, wohnung, gastName, d1, d2, "", "", "", "", ""]); } } } function extractIcalField(content, field) { var regex = new RegExp(field + "(?:;[^:]*)?:(.*(?:\\r?\\n [^\\r\\n]*)*)", "i"); var match = regex.exec(content); return match && match[1] ? match[1].replace(/\r?\\n/g, " ").replace(/\r?\n /g, "").trim() : ""; }
